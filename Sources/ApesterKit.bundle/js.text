function initAdvertisingParams(advertisingInfo) {
  // generate advertisingData
  var advertisingData = {iosDeviceInfo: JSON.parse(advertisingInfo)};
  // add apesterLoadCallback on window - this function will be called form the Apester Unit
  if (!window.apesterLoadCallback) {
    window.apesterLoadCallback = function() {
      window.APESTER.Init(advertisingData);
    }
  }
}
